6.4 Firmware: I2C Recovery (main.cpp)
This C++ snippet fixes the "Stuck Bus" issue on the ESP32-S3, a critical reliability fix for the Phenix Navigator. In mesh networks, a stuck I2C bus (due to a sensor glitch) can crash a node, breaking the mesh. This code manually "bit-bangs" 9 clock cycles to force any stuck slaves to release the SDA line before the hardware driver initializes.6


C++




#include "driver/gpio.h"
#include "driver/i2c_master.h"
#include "esp_log.h"
#include "rom/ets_sys.h"

#define I2C_SDA GPIO_NUM_8
#define I2C_SCL GPIO_NUM_9
#define TAG "I2C_RECOVERY"

/**
* @brief Bit-bangs the I2C bus to release stuck slaves.
* Standard "9-Clock" Recovery Sequence.
*/
void i2c_recover_bus() {
   ESP_LOGW(TAG, "Executing I2C Bus Recovery...");

   // 1. Configure pins as Open-Drain GPIO (NOT I2C yet)
   gpio_config_t io_conf = {};
   io_conf.intr_type = GPIO_INTR_DISABLE;
   io_conf.mode = GPIO_MODE_INPUT_OUTPUT_OD;
   io_conf.pin_bit_mask = (1ULL << I2C_SDA) | (1ULL << I2C_SCL);
   io_conf.pull_up_en = GPIO_PULLUP_ENABLE;
   gpio_config(&io_conf);

   // 2. Set Idle
   gpio_set_level(I2C_SDA, 1);
   gpio_set_level(I2C_SCL, 1);
   ets_delay_us(10);

   // 3. Check for Stuck Low SDA
   if (gpio_get_level(I2C_SDA) == 0) {
       ESP_LOGE(TAG, "SDA Stuck LOW. Clocking 9 cycles...");
       for (int i = 0; i < 9; i++) {
           gpio_set_level(I2C_SCL, 0); // Low
           ets_delay_us(10);
           gpio_set_level(I2C_SCL, 1); // High
           ets_delay_us(10);
           if (gpio_get_level(I2C_SDA) == 1) break; // Released
       }
   }
   
   // 4. Force STOP condition
   gpio_set_level(I2C_SDA, 0);
   ets_delay_us(10);
   gpio_set_level(I2C_SCL, 1);
   ets_delay_us(10);
   gpio_set_level(I2C_SDA, 1); // STOP
   
   ESP_LOGI(TAG, "Bus Recovery Complete.");
   
   // Pins must be reset before I2C driver init
   gpio_reset_pin(I2C_SDA);
   gpio_reset_pin(I2C_SCL);
}

extern "C" void app_main(void) {
   // ALWAYS recover before init
   i2c_recover_bus();

   // Initialize standard driver (10kHz safe mode)
   i2c_master_bus_config_t bus_config = {
     .i2c_port = I2C_NUM_0,
     .sda_io_num = I2C_SDA,
     .scl_io_num = I2C_SCL,
     .clk_source = I2C_CLK_SRC_DEFAULT,
     .glitch_ignore_cnt = 7,
     .flags = {.enable_internal_pullup = true }
   };

   i2c_master_bus_handle_t bus_handle;
   i2c_new_master_bus(&bus_config, &bus_handle);
}
