// Calculate Great Circle path between two nodes (Lat/Lon)
// Based on Charles Petzold's Great Circle logic 
function getGreatCirclePoints(startLat, startLon, endLat, endLon, numPoints) {
  // Convert Lat/Lon to Vector3
  const start = latLonToVector3(startLat, startLon, radius);
  const end = latLonToVector3(endLat, endLon, radius);
  
  // Slerp (Spherical Linear Interpolation)
  const points =;
  for(let i=0; i<=numPoints; i++) {
     const t = i / numPoints;
     const point = new THREE.Vector3().copy(start).lerp(end, t).normalize().multiplyScalar(radius * (1 + Math.sin(t * Math.PI) * 0.1)); // Add arc height
     points.push(point);
  }
  return points;
}

// TuningParticles.jsx
import React, { useRef, useMemo } from 'react';

// Vertex Shader: Morphs from Chaos (random) to Order (Sphere)
const vertexShader = `
 uniform float uProgress;
 uniform float uTime;
 attribute vec3 aPositionChaos;
 attribute vec3 aPositionOrder;
 varying float vSignal;

 // Simplex Noise function (simplified for brevity)
 //... [Include snoise3 function here]...

 void main() {
   // Add noise to the transition to simulate resistance
   float noise = snoise(vec3(aPositionChaos * 0.05 + uTime * 0.2));
   
   // Non-linear transition curve (Snap effect)
   float t = smoothstep(0.0, 1.0, uProgress + noise * 0.1);
   
   // Interpolate positions
   vec3 pos = mix(aPositionChaos, aPositionOrder, t);
   
   // Pass signal strength to fragment shader for coloring
   vSignal = t;
   
   vec4 mvPosition = modelViewMatrix * vec4(pos, 1.0);
   gl_Position = projectionMatrix * mvPosition;
   
   // Size attenuation based on depth
   gl_PointSize = (4.0 * (1.0 + t * 2.0)) * (100.0 / -mvPosition.z);
 }
`;

// Fragment Shader: Colors shift from Static (White/Grey) to Signal (Green/Gold)
const fragmentShader = `
 varying float vSignal;
 
 void main() {
   // Circle shape for particles
   vec2 uv = gl_PointCoord.xy - 0.5;
   if (length(uv) > 0.5) discard;
   
   // Color mixing: White -> Emerald (#10b981) -> Gold (#eab308)
   vec3 colorStatic = vec3(0.8, 0.8, 0.9);
   vec3 colorSignal = vec3(0.06, 0.72, 0.5); // Emerald
   vec3 colorLocked = vec3(0.91, 0.7, 0.03); // Gold
   
   vec3 finalColor = mix(colorStatic, colorSignal, vSignal);
   if(vSignal > 0.95) finalColor = mix(finalColor, colorLocked, (vSignal - 0.95) * 20.0);
   
   gl_FragColor = vec4(finalColor, 0.8);
 }
`;

export const TuningParticles = () => {
 const signalStrength = useGenesisStore(state => state.signalStrength);
 const count = 5000;
 const mesh = useRef();

 // Generate attributes once
 const { chaosPositions, orderPositions } = useMemo(() => {
   const chaos = new Float32Array(count * 3);
   const order = new Float32Array(count * 3);
   const sphereRadius = 15;
   
   for (let i = 0; i < count; i++) {
     // Chaos: Random box
     chaos[i*3] = (Math.random() - 0.5) * 100;
     chaos[i*3+1] = (Math.random() - 0.5) * 100;
     chaos[i*3+2] = (Math.random() - 0.5) * 100;
     
     // Order: Sphere surface
     const phi = Math.acos(-1 + (2 * i) / count);
     const theta = Math.sqrt(count * Math.PI) * phi;
     order[i*3] = sphereRadius * Math.cos(theta) * Math.sin(phi);
     order[i*3+1] = sphereRadius * Math.sin(theta) * Math.sin(phi);
     order[i*3+2] = sphereRadius * Math.cos(phi);
   }
   return { chaosPositions: chaos, orderPositions: order };
 },);

 const uniforms = useMemo(() => ({
   uProgress: { value: 0 },
   uTime: { value: 0 }

       transparent={true}

 );

// GlassBoxOverlay.jsx
import { useThree, useFrame } from '@react-three/fiber';
import { useEffect } from 'react';
import * as THREE from 'three';
import { useGenesisStore } from './store';

export const GlassBoxOverlay = () => {
 const active = useGenesisStore(state => state.glassBox);
 const { scene } = useThree();
 
 // Cache original materials
 useEffect(() => {
   scene.traverse((child) => {
     if (child.isMesh) {
       child.userData.originalMaterial = child.material;
     }
   });
 }, [scene]);

 useFrame(() => {
   scene.traverse((child) => {
     if (child.isMesh) {
       if (active) {
         // GLASS BOX MODE: Wireframe + Emissive
         if (!child.userData.glassMaterial) {
            child.userData.glassMaterial = new THREE.MeshBasicMaterial({
               color: 0x10b981, // Emerald
               wireframe: true,
               transparent: true,
               opacity: 0.3
            });
         }
         child.material = child.userData.glassMaterial;
       } else {
         // NORMAL MODE: Restore
         if (child.userData.originalMaterial) {
           child.material = child.userData.originalMaterial;
         }
       }
     }
   });
 });

 return null; // Logic only
};

This component forces a visual state change on the entire scene. When the user clicks "Glass Box," the texture-mapped reality vanishes, revealing the "Emerald" wireframe grid of the Delta topology.
