6.2 Step 2: The Custom Hook for Web Serial (The Bridge)
We need a robust hook to ingest the high-frequency data from the Phenix Navigator without re-rendering the component tree. We use a ref based approach or a transient state manager like Zustand. This bypasses the "React Tax" on performance.


JavaScript




// hooks/usePhenixSerial.js
import { useEffect, useRef } from 'react';
import { useStore } from '../store';

export function usePhenixSerial() {
 const portRef = useRef(null);
 const readerRef = useRef(null);
 const setDelta = useStore((state) => state.setDelta); // Transient update function

 const connect = async () => {
   try {
     // Request the port - triggers the browser permission prompt
     portRef.current = await navigator.serial.requestPort();
     await portRef.current.open({ baudRate: 115200 });
     
     const textDecoder = new TextDecoderStream();
     const readableStreamClosed = portRef.current.readable.pipeTo(textDecoder.writable);
     const reader = textDecoder.readable.getReader();
     readerRef.current = reader;

     // The Read Loop
     while (true) {
       const { value, done } = await reader.read();
       if (done) break;
       if (value) {
         try {
           // Parse the JSON payload from the ESP32
           // Expected format: {"d": 1} (delta) or {"b": 1} (button)
           // Using single letter keys reduces serial overhead
           const data = JSON.parse(value);
           if (data.d) setDelta(data.d); // Rotary delta (+1 / -1)
           if (data.b) useStore.getState().triggerMeasurement(); // Button press
         } catch (e) {
           // Ignore serial noise/incomplete packets
         }
       }
     }
   } catch (error) {
     console.error("Phenix Connection Failed:", error);
   }
 };

 return { connect };
}
