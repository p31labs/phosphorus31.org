      // Projection Logic
      const project = (v) => {
        // Rotation X
        let y1 = v.y * Math.cos(rx) - v.z * Math.sin(rx);
        let z1 = v.y * Math.sin(rx) + v.z * Math.cos(rx);
        // Rotation Y
        let x2 = v.x * Math.cos(ry) + z1 * Math.sin(ry);
        let z2 = -v.x * Math.sin(ry) + z1 * Math.cos(ry);
        
        // Perspective
        const fov = 4;
        const p = fov / (fov - z2);
        return { x: x2 * p * scale + cx, y: y1 * p * scale + cy };
      };


      const projected = vertices.map(project);


      // Draw Connection Lines (The 1/3 Constant Geometry)
      ctx.lineWidth = 2;
      ctx.lineCap = 'round';
      
      edges.forEach(([i, j]) => {
        const p1 = projected[i];
        const p2 = projected[j];
        
        // Gradient Stroke
        const grad = ctx.createLinearGradient(p1.x, p1.y, p2.x, p2.y);
        grad.addColorStop(0, '#fbbf24'); // Amber
        grad.addColorStop(0.5, '#f59e0b');
        grad.addColorStop(1, '#fbbf24');
        
        ctx.strokeStyle = grad;
        ctx.beginPath();
        ctx.moveTo(p1.x, p1.y);
        ctx.lineTo(p2.x, p2.y);
        ctx.stroke();


        // Draw Nodes
        ctx.fillStyle = '#1e293b';
        ctx.beginPath();
        ctx.arc(p1.x, p1.y, 4, 0, Math.PI * 2);
        ctx.fill();
        ctx.strokeStyle = '#22d3ee'; // Cyan
        ctx.stroke();

      ctx.arc(cx, cy, 3, 0, Math.PI * 2);

// --- HELPER BRAINS ---


function extractText(file) {
  const mime = file.getMimeType();
  // If it's a PDF or Image, we use the Drive API "OCR" trick
  if (mime === "application/pdf" || mime.includes("image")) {
    const resource = { title: file.getName(), mimeType: "application/vnd.google-apps.document" };
    const newFile = Drive.Files.insert(resource, file.getBlob(), { ocr: true });
    const doc = DocumentApp.openById(newFile.id);
    const text = doc.getBody().getText();
    Drive.Files.remove(newFile.id); // Delete the temp doc
    return text.toLowerCase();
  } else if (mime === "application/vnd.google-apps.document" || mime === "text/plain") {
    return DocumentApp.openById(file.getId()).getBody().getText().toLowerCase();
  }
  return "";
}


function calculateScores(text) {
  const scores = {};
  for (const [category, config] of Object.entries(AUTOMATON_CONFIG.CATEGORIES)) {
    scores[category] = 0;
    config.keywords.forEach(word => {
      if (text.includes(word)) scores[category]++;
    });
  }
  return scores;
}


function determineWinner(scores) {
  let maxScore = 0;
  let winner = null;
  for (const [cat, score] of Object.entries(scores)) {
    if (score > maxScore && score >= AUTOMATON_CONFIG.CATEGORIES[cat].threshold) {
      maxScore = score;
      winner = cat;
    }
  }
  return winner;
}


function performRedaction(text) {
  let cleanText = text;
  let found = false;
  AUTOMATON_CONFIG.REDACTION_PATTERNS.forEach(pat => {
    if (pat.regex.test(cleanText)) {
      found = true;
      cleanText = cleanText.replace(pat.regex, `[REDACTED ${pat.name}]`);
    }
  });
  return { cleanText: cleanText, foundSensitive: found };
}


function createMetaCard(originalFile, summaryText, category, parentFolder) {
  const note = DocumentApp.create(`NOTE_${originalFile.getName()}`);
  const body = note.getBody();
  body.appendParagraph(`ANALYSIS REPORT for ${originalFile.getName()}`).setHeading(DocumentApp.ParagraphHeading.HEADING1);
  body.appendParagraph(`Category Detected: ${category || "UNKNOWN"}`);
  body.appendParagraph(`Original ID: ${originalFile.getId()}`);
  body.appendHorizontalRule();
  body.appendParagraph("REDACTED TRANSCRIPT SUMMARY:");
  body.appendParagraph(summaryText.substring(0, 2000) + "..."); // First 2000 chars
  
  const file = DriveApp.getFileById(note.getId());
  file.moveTo(parentFolder);
}


function findFolderDeep(folder, name) {
  const files = folder.getFolders();
  while (files.hasNext()) {
    const f = files.next();
    if (f.getName() === name || f.getName().includes(name)) return f;
    const deep = findFolderDeep(f, name); // Recursive
    if (deep) return deep;
  }
  return null;
}

function getOrCreate(parent, name) {
  const f = parent.getFoldersByName(name);
  return f.hasNext() ? f.next() : parent.createFolder(name);
}


function createPath(parent, path) {
  let current = parent;
  path.split('/').forEach(part => current = getOrCreate(current, part));
}


function isDescendant(child, parent) {
  let current = child;
  while (current.getParents().hasNext()) {
    current = current.getParents().next();
    if (current.getId() === parent.getId()) return true;
    if (current.getId() === DriveApp.getRootFolder().getId()) return false;
  }
  return false;
}


function extractText(file) {
  const mime = file.getMimeType();
  if (mime === "application/pdf" || mime.includes("image")) {
    // OCR Trick
    const resource = { title: file.getName(), mimeType: "application/vnd.google-apps.document" };
    try {
      const newFile = Drive.Files.insert(resource, file.getBlob(), { ocr: true });
      const doc = DocumentApp.openById(newFile.id);
      const text = doc.getBody().getText();
      Drive.Files.remove(newFile.id);
      return text.toLowerCase();
    } catch (e) { return ""; }
  } else if (mime === "application/vnd.google-apps.document" || mime === "text/plain") {
    return DocumentApp.openById(file.getId()).getBody().getText().toLowerCase();
  }
  return "";
}


function calculateScores(text) {
  const scores = {};
  for (const [cat, config] of Object.entries(AUTOMATON_CONFIG.CATEGORIES)) {
    scores[cat] = 0;
    config.keywords.forEach(word => { if (text.includes(word)) scores[cat]++; });
  }
  return scores;
}


function determineWinner(scores) {
  let max = 0; let winner = null;
  for (const [cat, score] of Object.entries(scores)) {
    if (score > max && score >= AUTOMATON_CONFIG.CATEGORIES[cat].threshold) {
      max = score; winner = cat;
    }
  }
  return winner;
}


function performRedaction(text) {
  let clean = text; let found = false;
  AUTOMATON_CONFIG.REDACTION_PATTERNS.forEach(pat => {
    if (pat.regex.test(clean)) { found = true; clean = clean.replace(pat.regex, `[REDACTED ${pat.name}]`); }
  });
  return { cleanText: clean, foundSensitive: found };
}


function createMetaCard(originalFile, summaryText, category, parentFolder) {
  const note = DocumentApp.create(`NOTE_${originalFile.getName()}`);
  const body = note.getBody();
  body.appendParagraph(`ANALYSIS: ${originalFile.getName()}`).setHeading(DocumentApp.ParagraphHeading.HEADING1);
  body.appendParagraph(`Detected Category: ${category || "UNKNOWN"}`);
  body.appendParagraph(`Original ID: ${originalFile.getId()}`);
  body.appendParagraph("--- REDACTED SUMMARY ---");
  body.appendParagraph(summaryText.substring(0, 2000) + "...");
  DriveApp.getFileById(note.getId()).moveTo(parentFolder);
}


function findFolderDeep(folder, name) {
  const folders = folder.getFolders();
  while (folders.hasNext()) {
    const f = folders.next();
    if (f.getName() === name) return f;
    const deep = findFolderDeep(f, name);
    if (deep) return deep;
  }
  return null;
}

function getOrCreate(parent, name) {
  const f = parent.getFoldersByName(name);
  return f.hasNext() ? f.next() : parent.createFolder(name);
}


function createPath(parent, path) {
  let current = parent;
  path.split('/').forEach(part => current = getOrCreate(current, part));
}


function isDescendant(child, parent) {
  let current = child;
  while (current.getParents().hasNext()) {
    current = current.getParents().next();
    if (current.getId() === parent.getId()) return true;
    if (current.getId() === DriveApp.getRootFolder().getId()) return false;
  }
  return false;
}


function extractText(file) {
  // OCR Driver
  const mime = file.getMimeType();
  if (mime === "application/pdf" || mime.includes("image")) {
    const resource = { title: file.getName(), mimeType: "application/vnd.google-apps.document" };
    try {
      const newFile = Drive.Files.insert(resource, file.getBlob(), { ocr: true });
      const doc = DocumentApp.openById(newFile.id);
      const text = doc.getBody().getText();
      Drive.Files.remove(newFile.id);
      return text.toLowerCase();
    } catch (e) { return ""; }
  } else if (mime === "application/vnd.google-apps.document" || mime === "text/plain") {
    return DocumentApp.openById(file.getId()).getBody().getText().toLowerCase();
  }
  return "";
}


function calculateScores(text) {
  const scores = {};
  for (const [cat, config] of Object.entries(GENSYNC_CONFIG.CATEGORIES)) {
    scores[cat] = 0;
    config.keywords.forEach(word => { if (text.includes(word)) scores[cat]++; });
  }
  return scores;
}


function determineWinner(scores) {
  let max = 0; let winner = null;
  for (const [cat, score] of Object.entries(scores)) {
    if (score > max && score >= GENSYNC_CONFIG.CATEGORIES[cat].threshold) {
      max = score; winner = cat;
    }
  }
  return winner;
}


function performRedaction(text) {
  let clean = text; let found = false;
  GENSYNC_CONFIG.REDACTION_PATTERNS.forEach(pat => {
    if (pat.regex.test(clean)) { found = true; clean = clean.replace(pat.regex, `[REDACTED ${pat.name}]`); }
  });
  return { cleanText: clean, foundSensitive: found };
}


function createMetaCard(originalFile, summaryText, category, parentFolder) {
  const note = DocumentApp.create(`NOTE_${originalFile.getName()}`);
  const body = note.getBody();
  body.appendParagraph(`GENSYNC ANALYSIS: ${originalFile.getName()}`).setHeading(DocumentApp.ParagraphHeading.HEADING1);
  body.appendParagraph(`Signal Type: ${category || "UNKNOWN"}`);
  body.appendParagraph(`Source ID: ${originalFile.getId()}`);
  body.appendParagraph("--- BINARY LOGIC APPLIED (REDACTED) ---");
  body.appendParagraph(summaryText.substring(0, 2000) + "...");
  DriveApp.getFileById(note.getId()).moveTo(parentFolder);
}


function findFolderDeep(folder, name) {
  const folders = folder.getFolders();
  while (folders.hasNext()) {
    const f = folders.next();
    if (f.getName() === name) return f;
    const deep = findFolderDeep(f, name);
    if (deep) return deep;
  }
  return null;
}
